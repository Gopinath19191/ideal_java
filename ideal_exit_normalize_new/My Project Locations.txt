SELECT `Project`.project_code,
       `Project`.project_name,
       `Customer`.customer_name,
       `RBU`.name AS 'RBU',
       `SBU`.name AS 'SBU',
       `Employee`.`employee_number`,
       concat(`Employee`.`first_name`, " ", `Employee`.`last_name`)
          AS 'employee_name',
       CASE ProjectTeamAllocation.working_place
          WHEN "s" THEN "Onsite"
          WHEN "o" THEN "Offshore"
       END
          AS "work_location",
       Role.role_description AS 'Role',
       concat(`emp2`.`first_name`, " ", `emp2`.`last_name`) AS 'pm_name',
       concat(`emp3`.`first_name`, " ", `emp3`.`last_name`) AS 'sales_person',
       CASE `Project`.`pricing`
          WHEN "fb" THEN "FB"
          WHEN "tm" THEN "T&M"
       END
          AS "Project Status",
       `BillingAdvice`.billing_month,
       `BillingAdvice`.billing_year,
       `BillingAdvice`.billable_effort,
       CASE `Project`.efforts_uom
          WHEN "h" THEN "Hour"
          WHEN "d" THEN "Day"
          WHEN "w" THEN "Week"
          WHEN "m" THEN "month"
       END
          AS "UOM",
       `ProjectTeamAllocation`.billing_rate,
       `Currency`.currency,
       `BillingAdvice`.billable_amount
  FROM `billing_advices` AS `BillingAdvice`
       LEFT JOIN `projects` AS `Project`
          ON (`BillingAdvice`.`project_id` = `Project`.`id`)
       LEFT JOIN `employees` AS `Employee`
          ON (`BillingAdvice`.`employee_id` = `Employee`.`id`)
       LEFT JOIN `employees` AS emp2
          ON (emp2.id = Project.project_manager)
       LEFT JOIN `employees` AS emp3
          ON (emp3.id = Project.sales_person)
       LEFT JOIN `project_team_allocations` AS `ProjectTeamAllocation`
          ON (`ProjectTeamAllocation`.`project_id` =
                 `BillingAdvice`.`project_id`
              AND `ProjectTeamAllocation`.`employee_id` =
                     `BillingAdvice`.`employee_id`)
       LEFT JOIN `roles` AS `Role`
          ON (`Role`.`id` = ProjectTeamAllocation.role_id)
       LEFT JOIN `currencies` AS `Currency`
          ON (`Currency`.`id` = `Project`.`currency_id`)
       LEFT JOIN `customers` AS `Customer`
          ON (`Customer`.`id` = `Project`.`customer_id`)
       LEFT JOIN `company_structures` AS `RBU`
          ON (`Project`.`project_rbu` = `RBU`.`id`)
       LEFT JOIN `company_structures` AS `SBU`
          ON (`Project`.`project_sbu` = `SBU`.`id`)
 WHERE     `BillingAdvice`.`deleted` = '0'
       AND `ProjectTeamAllocation`.`deleted` = '0'
       AND BillingAdvice.billing_year = 2011
       AND BillingAdvice.billing_month = 06
       AND `BillingAdvice`.`status` IN ('a', 'm')
       AND `Project`.project_status IN ('e') 
       AND CONCAT(BillingAdvice.billing_year, BillingAdvice.billing_month) BETWEEN EXTRACT(
                                                                                      YEAR_MONTH FROM ProjectTeamAllocation.
                                                                                                      start_date)
                                                                               AND EXTRACT(
                                                                                      YEAR_MONTH FROM ProjectTeamAllocation.
                                                                                                      end_date)
GROUP BY `Project`.`id`,
         `Employee`.`id`,
         `BillingAdvice`.`billing_month`,
         `BillingAdvice`.`billing_year`

SELECT appraisee.employee_number AS 'Appraisee Number',CONCAT(appraisee.first_name,' ',appraisee.last_name) AS 'Appraisee Name',
appraiser.employee_number AS 'Appriaser Number',CONCAT(appraiser.first_name,' ',appraiser.last_name) AS 'Appraiser Name',
reviewer.employee_number AS 'Reviewer Number',CONCAT(reviewer.first_name,' ',reviewer.last_name) AS 'Reviewer Name',
CASE qea.managerSurvey WHEN 1 THEN "Yes" WHEN 0 THEN "No" ELSE "Not Applicable" END AS "Manager Discussion",
qea.appraiserComments AS 'Appraiser Comments',
qea.areasOfImprovement AS 'Area of Improvement',
qea.technologyTraining AS 'Technology Training',
qea.softskillTraining AS 'Soft Skill Training',
CASE qea.submitStatus 
WHEN 0 THEN "Appraisee Not Submitted" 
WHEN 2 THEN "Appraiser Not Submitted" 
WHEN 4 THEN "Reviewer Not Submitted"
WHEN 6 THEN "Final Reviewer Not Submitted"
WHEN 8 THEN "Hr Not Submitted"
ELSE "Not Applicable" END AS "Status",
cs.name AS 'Practice Group'
FROM qpd_eligible_associates qea 
LEFT JOIN employees AS appraisee ON(appraisee.id=qea.appraiseeId)
LEFT JOIN employees AS appraiser ON(appraiser.id=qea.appraiserId)
LEFT JOIN employees AS reviewer ON(reviewer.id=qea.reviewerId)
LEFT JOIN company_structures AS cs ON(cs.id=appraiser.structure_name)
WHERE quarter =4
AND year =2011 AND appraisalTriggered=1
